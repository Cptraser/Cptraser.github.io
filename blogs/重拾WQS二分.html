#include <cstdio>
#include <cctype>
#include <algorithm>
#define V 2005
using namespace std;

const double eps=1e-10;
​
int N,a,b,Ca,Cb;
double A[V],B[V],Ans,CA,CB;
int Work(double Cost){
    CB=Cost;Ca=Cb=0;Ans=0;
    for(int i=1;i<=N;i++){
        double v=0;int opt=0;
        if(A[i]-CA>v)opt=1,v=A[i]-CA;
        if(B[i]-CB>v)opt=2,v=B[i]-CB;
        if(A[i]+B[i]-A[i]*B[i]-CA-CB>v)opt=3,v=A[i]+B[i]-A[i]*B[i]-CA-CB;
        if(opt==1||opt==3)Ca++;
        if(opt==2||opt==3)Cb++;
        Ans+=v;
    }return Cb;
}
int Check(double Cost){
    double L=0,R=1.0;CA=Cost;
    while(R-L>eps){
        double Mid=(L+R)/2;
        if(Work(Mid)<=b)R=Mid;
        else L=Mid;
    }return Work(CB=R),Ca;
}
int main()
{
    scanf("%d%d%d",&N,&a,&b);
    for(int i=1;i<=N;i++)scanf("%lf",&A[i]);
    for(int i=1;i<=N;i++)scanf("%lf",&B[i]);
    double L=0,R=1.0;
    while(R-L>eps){
        double Mid=(L+R)/2;
        if(Check(Mid)<=a)R=Mid;
        else L=Mid;
    }
    Check(R),Ans+=CA*a+CB*b;
    printf("%.12lf",Ans);
    return 0;
}
