double G(void){
    for(i=1;i<=7;i++){
        for(f[i]=Inf,j=1;j<=N;j++){
            if(Pr[i][j])f[i]=std::min(f[i],(f[i-1]+C[j]-(1-Pr[i][j])*f[Ls[i][j]])/Pr[i][j]);
        }
    }
    return f[7];
}
